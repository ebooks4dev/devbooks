CREATE FUNCTION [dbo].[ToLocalTime]
(
	-- Add the parameters for the function here
	@utcTime DATETIME
)
RETURNS DATETIME
AS
BEGIN	
	DECLARE @value DATETIME
	SET @value = DATEADD(HOUR, 7, @utcTime)
	RETURN @value
END
GO
